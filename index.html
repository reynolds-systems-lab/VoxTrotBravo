function calculateArcPoints(p1, p2, ctrlL, ctrlR, outPoints, fixedSteps = 0) { const dist = p1.distanceTo(p2); if (dist < 0.001 || !ctrlL || !ctrlR) return; // Use existing globals to prevent GC pressure and ReferenceErrors const lineAxis = currentHandVec.subVectors(p2, p1).normalize(); // Capture world rotation to keep control intuitive relative to view const worldGroupInvQuat = tempQuat2; worldGroup.getWorldQuaternion(worldGroupInvQuat).invert(); const getHandBulge = (ctrl, targetVec) => { // Get controller "Up" in local space targetVec.set(0, 1, 0).applyQuaternion(ctrl.quaternion).applyQuaternion(worldGroupInvQuat); // Project onto plane perpendicular to line const dot = targetVec.dot(lineAxis); targetVec.subScaledVector(lineAxis, dot); // Safety check: if hand is parallel to the line, use world "Up" if (targetVec.lengthSq() < 0.0001) { targetVec.set(0, 1, 0).applyQuaternion(worldGroupInvQuat); const dotFallback = targetVec.dot(lineAxis); targetVec.subScaledVector(lineAxis, dotFallback); } // Final safety: ensure we never normalize a zero-length vector (prevents NaNs) if (targetVec.lengthSq() > 0.00001) { targetVec.normalize().multiplyScalar(arcBulge * dist * 0.5); } else { targetVec.set(0, 0, 0); } return targetVec; }; // Reuse existing vectors for control point math const bulgeVecL = getHandBulge(ctrlL, tempPXZ); const bulgeVecR = getHandBulge(ctrlR, tempCXZ); const cp1 = tempPos.copy(p1).add(bulgeVecL); const cp2 = tempStart.copy(p2).add(bulgeVecR); const steps = fixedSteps > 0 ? fixedSteps : Math.min(100, Math.max(10, Math.ceil(dist / (VOXEL_SIZE * 2)))); outPoints.length = 0; for (let i = 0; i <= steps; i++) { const t = i / steps; const omt = 1 - t; const omt2 = omt * omt; const t2 = t * t; // Standard Cubic Bezier calculation const p = new THREE.Vector3(); p.addScaledVector(p1, omt2 * omt); p.addScaledVector(cp1, 3 * omt2 * t); p.addScaledVector(cp2, 3 * omt * t2); p.addScaledVector(p2, t2 * t); outPoints.push(p); } }
